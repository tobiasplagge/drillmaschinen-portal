API-Dokumentation

Alle API-Endpunkte unter /api/v1/ · Authentifizierung via HTTP Basic Auth + Bearer Token Header.

# Authentifizierung
POST /api/v1/auth/token
Body: { "username": "...", "password": "..." }
Response: { "token": "eyJ...", "expires_at": "..." }
# Neue Fahrt übermitteln
POST /api/v1/trips
Headers: Authorization: Bearer {token}
Body: { "machine_id": "DR-001", "field_name": "Nordfeld", "started_at": "...",
        "gps_points": [...], "events": [...] }
# GPS-Punkt zur laufenden Fahrt hinzufügen (Live-Übertragung)
POST /api/v1/trips/{trip_id}/points
Body: { "lat": 51.4823, "lon": 9.2145, "timestamp": "...", "speed": 7.2, "working": true }
# Ereignis (Störung/Gebläse) melden
POST /api/v1/trips/{trip_id}/events
Body: { "type": "fault|blower|info", "message": "Verstopfung Reihe 4",
        "lat": 51.4823, "lon": 9.2145, "timestamp": "..." }
# Fahrt abschließen
PATCH /api/v1/trips/{trip_id}/finish
Body: { "ended_at": "...", "area_ha": 18.4 }
